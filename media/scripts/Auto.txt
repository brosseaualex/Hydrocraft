/************************RECIPES************************/	

recipe Attach Clamps to Car Battery
   {
	SkillRequired:Electricity=1,
	destroy CarBattery1/CarBattery2/CarBattery3,
	HCElectricclamp=2,  
        Result:HCCarbattery,
        Time:50,
       	Category:Electrical,
       	OnGiveXP:HCElectricity_OnGiveXP,
    }

recipe Repair Bicycle
   {
	SkillRequired:Mechanics=1,  
	HCJunkbicycle,
	HCBicyclewheel,
	DuctTape=4,
	HCRubbercement=4,
        keep Screwdriver,
	keep HCPliers,
	keep HCBicyclepump,
        Result:HCBicycle,
        Time:500,
        NeedToBeLearn:true,
       	Category:Engineer,
    	OnGiveXP:HCMechanics_OnGiveXP,
    }

recipe Attach Baskets to Bicycle

	HCStrawbasket=2,
	Rope,

        Time:50,
